#!/bin/bash

workdir=`pwd`



[ -e NoFoundFiles.txt ] && rm NoFoundFiles.txt

[ -e FoundFiles.txt ] && rm FoundFiles.txt



files_count=$(ls -l *.vcf | wc -l)

if [[ $1 == "-d" ]];then

cut -f 2 *.vcf|sort -n|uniq -c|grep -P "\b${files_count}\b"

elif [[ $1 == "-c" ]];then


cut -f 2 *.vcf|sort -n|uniq -c|grep -P "\b${files_count}\b"|awk '{ print $2}'|grep -P "^(\d+)"|tr -d '[:blank:]'|awk -v group=$2 -v counted=$files_count '{ print $1,"\t",counted, "\t", group}'

fi


if [[ $1 == "-i" ]];then


#while read -r line ; do

#columns=$(cut -f2 $line)


posArr=()

pos=$(cut -f2 $2)

posArr+=(${pos})

NotFoundIn=()

FoundIn=()



for i in "${posArr[@]}"

do

count_words=$(grep -P "\t$i\t" *.vcf|wc -l)



if [[ $count_words == $files_count ]]; then

printf "%s\t%s\t%s\tOK\n" $i $files_count $count_words

((goodCount++))

elif [[ $count_words == 0 ]]; then

printf "%s\t%s\tNOT_FOUND\n" $i $count_words

elif [[ $count_words != $files_count ]]; then

printf "%s\t%s\t%s\tFAIL\n" $i $files_count $count_words

NotFoundIn+=($(grep -P "\t$i\t" *.vcf -L))

FoundIn+=($(grep -P "\t$i\t" *.vcf -l))

((failCount++))

NotFoundInRes=$(printf ",%s" "${NotFoundIn[@]}")

NotFoundInRes=${NotFoundInRes:1}

FoundInRes=$(printf ",%s" "${FoundIn[@]}")

FoundInRes=${FoundInRes:1}



echo -e $i"\t"NOT FOUND IN $(( $files_count-$count_words)) FROM $files_count "\n" $NotFoundInRes >>NoFoundFiles.txt

echo "--------------------------------------------------------" >>NoFoundFiles.txt

echo -e $i"\t"FOUND IN $count_words FROM $files_count "\n" $FoundInRes >>FoundFiles.txt

echo "--------------------------------------------------------" >>FoundFiles.txt



fi


done


echo "-----------------------"

printf "OK:%s\t FAILS: %s\n" $goodCount $failCount


fi
