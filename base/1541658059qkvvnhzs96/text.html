4. распаковал


gunzip *.gz


5. собрал геномы в одном мультифаста файле:


cat *.fna > multifasta.fna



6. сравнил геномы при помощи nucmer.


nucmer H37rv.fasta mutlifasta.fna


получится файл: out.delta


7.экстрагируем координаты пересекающихся регионов


show-coords -r -T -o  out.delta >result

9. линкуем efetch и edirect.pl в рабочую папку


10. используем скрипт: getSeqByNucmerCoords.sh



#!/usr/bin/env bash



if [ -z "$1" ];

then

echo "Укажи файл для парсинга!"

else


while IFS="\t" read -r data

do

arr=($data)

./efetch -db nuccore -format fasta -id ${arr[8]} -chr_start ${arr[2]} -chr_stop ${arr[3]}

done < $1


fi


./getSeqByNucmerCoords.sh testRegion.txt > testRegionMultifasta.fna
